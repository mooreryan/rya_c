90 {

93  char* buffer = NULL;
94 
95  do {
96  *str_ptr = (char*)realloc(buffer, (size_t)buffer_size);
97  if (*str_ptr == NULL) {
98  rya_free(buffer);
99  return -1;
100  }
101 
102  // now all the chars are non-null, we'll need this to check later.
103  memset(*str_ptr, '?', (size_t)buffer_size);
104 
105  // If we're on a platform where this is a deep copy, that's good.
106  // If not, hope that you don't have args that require deep copying
107  // as vsnprintf isn't guaranteed not to change things.
108  va_list args_copy;
109  va_copy(args_copy, args);

117  }
118  // Insufficient storage space is available.

127  // hack it. We know vsnprintf() nul terminates at the
128  // buffer_size'th position in the buffer. So if we find the nul
129  // char before the size of the buffer we know it was at least sort
130  // of successful.

136  // memset the buffer to '?' and put the null terminator at
137  // buffer_size'th index. If for some reason, vsnprintf on error
138  // puts a null char in the buffer somewhere before buffer size,
139  // the result won't be what you want possibly, but it won't be
140  // invalid as it will fit inside of str_ptr properly.

143  // TODO clean up str_ptr and buffer.

146  }

148  // (v)snprintf functions should return the number of chars

151  }
